Element.py

''' This module represents object oriented structure in Python. The Element
    class is the smallest unit of matter, with methods that assist with
    its string and integer representations. '''

class Element(object): # "new" style for writing objects, inherit from "object"
  def __init__(self, atno, symbol, mass):
    self.atno = atno # atomic number
    self.symbol = symbol # atomic symbol
    self.mass = mass # representation of atomic mass
  
  '''
  Getter method for current atomic number value
  @param  self This current object
  @return      The current object's atno value '''
  def atomic_number(self):
    return self.atno
  
  '''
  Getter method for current atomic symbol representation
  @param  self This current object
  @return      The current element's symbol '''
  def get_symbol(self):
    return self.symbol
  
  '''
  Getter method for current atomic mass value
  @param  self This current object
  @return      The current object's mass value '''
  def get_mass(self):
    return self.mass

  def set_mass(self, new_mass):
    self.mass = new_mass
  
  '''
  String overload ("repr - esentation")
  @param  self This current object
  @return      A string representation of element '''
  def __repr__(self):
    return "{0}: Atomic #{1}, Mass {2} g/mol".format(self.symbol, self.atno, self.mass)

from element_data import elements

while True:
    user_input = input("Element: ")
    if user_input in elements:
        print(elements[user_input])
    elif user_input == "exit":
        break
    else:
        print("Not a valid input! Please try again.")
